public function PictureWheel(pictures:Vector.<Bitmap>, rect:Rectangle = null, baseMap:Bitmap = null, quantity:uint, dragOffset:uint = 100, sizeDecay:Number = 0.8, distanceDecay:Number = 0.3, filters:Array = null)

build a new PictureWheel

Parameters
pictures:Vector.<Bitmap> — pictures
 
rect:Rectangle (default = null) — the display area of the picture on the base map
 
baseMap:Bitmap (default = null) — base map
 
quantity:uint (default = NaN) — max quantity of display(Tip: max is 5)
 
dragOffset:uint (default = 100) — swap picture if the distance of mouse move larger than this value
 
sizeDecay:Number (default = 0.8) — size decay rate of the bigger picture to the smaller picture
 
distanceDecay:Number (default = 0.3) — distance decay rate of the bigger picture to the smaller picture
 
filters:Array (default = null) — filters of the side pictures

var EBasemap:Class;
[Embed(source = "../lib/embed/movie.png")]
var EMovie:Class;

var p:PictureWheel = new PictureWheel(
    Vector.《Bitmap》([new EMovie, new EMovie, new EMovie, new EMovie, new EMovie, new EMovie, new EMovie]), 
    new Rectangle(0, 0, 470, 270), 
    new EBasemap, 
    7,
    100,
    .7,
    .2,
    [new BlurFilter()]
);
p.pageDot = new PageDot(true);
addChild(p);
p.addEventListener(Event.SELECT, function(e:Event):void { trace(e.target.clickIndex); } );
