模块功能：

程序编制：

trace(Chinese.convertChar("中国人").toUpperCase(), Chinese.convertString("中国人"));

var arr1:Array = ["中国人", "日本人", "韩国人"];
trace(arr1);
trace(Chinese.sort(arr1));

var arr2:Array = [{label:"中国人2"}, {label:"日本人2"}, {label:"韩国人2"}];
trace(arr2);
trace(Chinese.sort(arr2, "label"));
